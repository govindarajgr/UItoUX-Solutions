SET ANSI_NULLS ON
GO
SET QUOTED_IDENTIFIER ON
GO
CREATE PROCEDURE [Product].[FetchHomepageDetails]
AS
BEGIN
    SET NOCOUNT ON;

    BEGIN TRY
        SET NOCOUNT ON;

    -- Table[0] To get the Brand Details
    SELECT 
        BrandId,
        BrandName,
        BrandLogo
    FROM 
        [Product].[BrandDetails]
    WHERE 
        IsDeleted = 0;

    -- Table[1] To get the Product Details
    SELECT 
        PD.ProductName,
        PT.ProductType,
        PD.ModelName,
        PD.ImageName,
        CASE 
            WHEN PD.DiscountPrice > 0 THEN PD.DiscountPrice 
            ELSE PD.OrginalPrice 
        END AS Price,
        CASE 
            WHEN PD.DiscountPrice > 0 THEN PD.OrginalPrice 
            ELSE 0 
        END AS OrginalPrice,
        ISNULL(PR.ProductRating, 0) AS ProductRating,
        ISNULL(PR.Rating, 0) AS Rating
    FROM 
        [Product].[ProductTypeDetails] PTD
    LEFT JOIN 
        [Product].[ProductDetails] PD ON PTD.ProductId = PD.ProductId
    LEFT JOIN 
        [Support].[ProductType] PT ON PTD.ProductTypeId = PT.ProductTypeId
    LEFT JOIN 
        (
            SELECT 
                ProductId,
                COUNT(ProductRatingId) AS ProductRating,
                AVG(Rating) AS Rating 
            FROM 
                [Product].[ProductRatingDetails]
            WHERE 
                IsDeleted = 0
            GROUP BY  
                ProductId
        ) PR ON PTD.ProductId = PR.ProductId
    WHERE 
        PTD.IsDeleted = 0  
        AND PD.IsDeleted = 0 
        AND PT.IsDeleted = 0
    GROUP BY 
        PD.ProductId,
        PD.ProductName,
        PT.ProductType,
        PD.ImageName,
        PD.ModelName,
        PD.DiscountPrice,
        PD.OrginalPrice,
        PR.ProductRating,
        PR.Rating,
        PT.DisplayOrder
    ORDER BY 
        PT.DisplayOrder, 
        PD.ProductId;

    -- Table[3] To get the Slide Details
    SELECT 
        SlideId,
        Header,
        Description,
        Notes,
        SlideImage AS ImageName
    FROM 
        [Product].[SlideDetails]
    WHERE 
        IsDeleted = 0;

    -- Table[4] To get the Banner Details
    SELECT 
        BannerId,
        BannerDescription,
        BannerNotes,
        BannerImages AS ImageName
    FROM 
        [Product].[BannerDetails]
    WHERE 
        IsDeleted = 0;
    END TRY
    BEGIN CATCH
        -- Error handling code
        DECLARE @ErrorMessage NVARCHAR(4000);
        DECLARE @ErrorSeverity INT;
        DECLARE @ErrorState INT;

        SELECT 
            @ErrorMessage = ERROR_MESSAGE(),
            @ErrorSeverity = ERROR_SEVERITY(),
            @ErrorState = ERROR_STATE();

        RAISERROR (@ErrorMessage, @ErrorSeverity, @ErrorState);
    END CATCH
END;
Go
